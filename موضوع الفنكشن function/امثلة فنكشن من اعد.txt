// factorial calculator - function
#include <iostream>
using namespace std;
int factorial (int a)
{
	  if (a > 1)
	   return (a * factorial (a-1));
	  else
	   return 1;
}

int main ()
{
  int number;
  cout << "Please enter a number: ";
  cin >> number;
  cout << number << "! = " << factorial (number) << endl;
  cout<< "\nDesgin: Rafea Fheely\n";
  return 0;
}

  cout<< "\nDesgin: Rafea Fheely\n";
  return ;
}

{
	int n;
